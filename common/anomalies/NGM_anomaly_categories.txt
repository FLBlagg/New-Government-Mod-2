# 1.欠落した生態系 missing_ecosystem Ⅳ
# ①アノマリー表記等（不毛な惑星等のみ）
# この星には大規模な生態系の跡が残っているが、不自然な生態系が存在しており、生命の終焉への道をたどっている。
ngm_missing_ecosystem_category = {
	# picture = GFX_evt_atmospheric_flight
	level = 4

	spawn_chance = {
		modifier = {
			add = 3
			uninhabitable_regular_planet = yes
		}
	}

	max_once_global = yes

	on_success = ngm_anomaly.1
}
