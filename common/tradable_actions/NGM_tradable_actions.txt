# 志向の矯正
trade_action_ngm_ethics_force = {
	fire_and_forget = no
	potential = {
		has_overlord_dlc = yes
		has_overlord = FROM
		NOT = {
			is_action_active = {
				action = trade_action_ngm_ethics_force
				with_country = FROM
			}
		}
	}
	active = {
		has_overlord = FROM
	}
	on_traded_effect = {
		add_modifier = {
			modifier = ngm_ethics_force_loyalty_1
		}
	}
	on_deal_ended_sender_effect = {
		remove_modifier = ngm_ethics_force_loyalty_1
	}
	ai_weight = {
		weight = 2
	}
}
