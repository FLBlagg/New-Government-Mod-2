		desc = POP_ATTRACTION_RULER_FACTION_POS

	}

	pop_attraction_tag = {
		desc = POP_ATTRACTION_SLAVERY_POS
		trigger = {
			any_enslaved_species = {

		}
	}

		desc = POP_ATTRACTION_XENO_SLAVES_POS
		trigger = {
			any_enslaved_species = {

			}
			NOT = {
				any_enslaved_species = {
					is_same_species = root
					is_robotic = no
				}
			}
		}
	}

		desc = POP_ATTRACTION_CONSUMER_GOODS_NEG
