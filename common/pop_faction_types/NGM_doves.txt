##############################
# Doves Faction
##############################

NGM_doves = {
	election_header = "GFX_faction_header_yellow"
	guiding_ethic = ethic_neutralism

	unique = yes
	support_multiplier = 1

	resources = {
		category = pop_factions
		produces = {
			unity = 0.5
		}
	}

	is_potential = {
		years_passed > 10
		has_encountered_other_species = yes
		NOT = { has_ethic = ethic_gestalt_consciousness }
		NOT = { has_country_flag = suppress_vanilla_factions }
	}

	parameters = {
		empire = {
			type = country
			valid_objects = {
				is_same_value = root
			}
		}
	}

	can_join_pre_triggers = {
		is_enslaved = no
		is_being_purged = no
		is_robot_pop = no
	}

	can_join_faction = {
		NOR = { 
			has_trait = trait_nerve_stapled
			has_trait = trait_zombie
		}
		OR = {
			has_ethic = ethic_neutralism
			has_ethic = ethic_fanatic_neutralism
		}
	}

	attraction = {
		base = 150

		modifier = {
			factor = 1.25
			exists = from
			from = {
				exists = leader
				leader = { leader_class = ruler }
			}
		}

		modifier = {
			factor = 0.75
			exists = from
			from = {
				NOT = { exists = leader }
			}
		}
	}

	leader = {
		base = 100

		modifier = {
			factor = 0.0
			is_event_leader = yes
		}
	}
	demand = {
		title = "FACTION_OUTRAGED_GC"
		unfulfilled_title = "FACTION_OUTRAGED_GC"
		desc = "FACTION_OUTRAGED_GC_DESC"
		fulfilled_effect = -20
		unfulfilled_effect = -0.001
		potential = {
			exists = owner
			owner = {
				has_country_flag = faction_neutralism_outraged_gc
			}
		}
		trigger = {
			owner = {
				has_country_flag = faction_neutralism_outraged_gc
			}
		}
	}
	demand = {
		# 強力なる孤立
		title = "SUPREMACIST_ISOLATIONIST_NO_FEDERATION"
		unfulfilled_title = "SUPREMACIST_ISOLATIONIST_FEDERATION"
		desc = "SUPREMACIST_ISOLATIONIST_NO_FEDERATION_DESC"

		unfulfilled_effect = -25
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 0
			}
		}

		trigger = {
			owner = {
				has_federation = no
			}
		}
	}
	demand = {
		# 多数の不可侵条約
		title = "ISOLATIONIST_NON_AGGRESSION3"
		unfulfilled_title = "ISOLATIONIST_FEW_NON_AGGRESSION3"
		desc = "ISOLATIONIST_NON_AGGRESSION_DESC3"

		fulfilled_effect = 15
		unfulfilled_effect = -0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 2
				count_playable_country = {
					limit = {
						OR = {
							has_non_aggression_pact = root.owner
							has_association_status = root.owner
						}
					}
					count > 2
				}
			}
		}

		trigger = {
			owner = {
				count_playable_country = {
					limit = {
						OR = {
							has_non_aggression_pact = root.owner
							has_association_status = root.owner
						}
					}
					count > 2
				}
			}
		}
	}
	
	demand = {
		# "平和"
		title = "PROSPERITY_PEACE"
		unfulfilled_title = "PROSPERITY_NO_PEACE"
		desc = "PROSPERITY_PEACE_DESC"

		fulfilled_effect = 5
		unfulfilled_effect = -10

		potential = {
			exists = owner
		}

		trigger = {
			owner = {
				is_at_war = no
			}
		}
	}
	demand = {
		# 非暴力
		title = "PROSPERITY_NOT_AGGRESSOR"
		unfulfilled_title = "PROSPERITY_AGGRESSOR"
		desc = "PROSPERITY_NOT_AGGRESSOR_DESC"

		unfulfilled_effect = -30
		fulfilled_effect = 0.001

		potential = {
			exists = owner
			owner = {
				num_communications > 2
			}
		}

		trigger = {
			owner = {
				NOR = {
					has_country_flag = entered_war_as_attacker
					any_war = { is_war_participant = { who = prev side = attackers } }
				}
			}
		}
	}
	demand = {
		# あらゆる難民を歓迎する
		title = "XENOIST_REFUGEES_WELCOME"
		unfulfilled_title = "XENOIST_REFUGEES_UNWELCOME"
		desc = "XENOIST_REFUGEES_WELCOME_DESC"

		fulfilled_effect = 10
		unfulfilled_effect = -10

		potential = {
			exists = owner
			owner = {
				any_relation = {
					has_communications = root.owner
					is_country_type = default
				}
			}
		}

		trigger = {
			owner = {
				has_policy_flag = refugees_allowed
			}
		}
	}

	on_create = {
		pop_faction_event = {
			id = factions.1
		}
	}
	on_destroy = {
		if = {
			limit = {
				has_modifier = promoted_neutralism
			}
			remove_modifier = promoted_neutralism
		}
		if = {
			limit = {
				has_modifier = suppressed_neutralism
			}
			remove_modifier = suppressed_neutralism
		}
	}
	actions = {
		embrace_faction = {
			title = "EMBRACE_FACTION"
			description = "EMBRACE_FACTION_DESC"

			cost = {
				unity = 5000
			}

			potential = {
				exists = owner
				owner = {
					OR = {
						is_subject = no
						NOT = { is_subject_type = dominion }
					}
				}
			}

			valid = {
				custom_tooltip = {
					fail_text = EMBRACE_FACTION_COOLDOWN
					parameter:empire = {
						NOT = { has_modifier = embraced_faction_timer }
					}
				}
				support > 0.20
				parameter:empire = {
					NOT = {
						has_ethic = "ethic_fanatic_neutralism"
					}
				}
			}
			effect = {
				add_modifier = {
					modifier = embraced_faction
					days = 3600
				}
				parameter:empire = {
					shift_ethic = ethic_neutralism
					hidden_effect = {
						add_modifier = {
							modifier = embraced_faction_timer
							days = 3600
						}
						every_pop_faction = {
							limit = {
								NOT = {
									is_same_value = root
								}
							}
							add_modifier = {
								modifier = embraced_another_faction
								days = 3600
							}
						}
					}
				}
				hidden_effect = {
					save_event_target_as = TargetFaction
					parameter:empire = {
						every_relation = {
							limit = {
								is_ai = no
								is_country_type = default
								has_communications = prev
								has_intel_level = {
									who = prev
									category = government
									level >= 1
								}
							}
							country_event = {
								id = factions.2000
							}
						}
					}
				}
			}
			ai_weight = {
				base = 1
				modifier = {
					factor = 0
					support < 0.5
					owner = {
						has_ethic = ethic_neutralism
					}
				}
				modifier = {
					factor = 0
					owner = {
						has_valid_civic = civic_fanatic_purifiers
					}
				}
				modifier = {
					factor = 0
					NOT = {
						owner = {
							exists = ruler
							ruler = {
								leader_of_faction = NGM_doves
							}
						}
					}
				}
				modifier = {
					factor = 0
					owner = {
						count_pop_faction = {
							count < 4
						}
					}
				}
			}
		}
		promote_faction = {
			title = "PROMOTE_FACTION"
			description = "PROMOTE_FACTION_DESC"
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_neutralism
						has_modifier = promoted_neutralism
					}
				}
			}
			effect = {
				parameter:empire = {
					add_modifier = {
						modifier = promoted_neutralism
						days = -1
					}
				}
			}
			ai_weight = {
				base = 0
			}
		}
		cancel_promote_faction = {
			title = "CANCEL_PROMOTE_FACTION"
			description = "CANCEL_PROMOTE_FACTION_DESC"
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = promoted_neutralism
				}
			}
			effect = {
				parameter:empire = {
					remove_modifier = promoted_neutralism
				}
			}
			ai_weight = {
				base = 0
			}
		}
		suppress_faction = {
			title = "SUPPRESS_FACTION"
			description = "SUPPRESS_FACTION_DESC"
			potential = {
				exists = owner
				parameter:empire = {
					NOR = {
						has_modifier = suppressed_neutralism
						has_modifier = promoted_neutralism
					}
				}
			}
			effect = {
				add_modifier = {
					modifier = suppressed_faction
					days = -1
				}
				parameter:empire = {
					add_modifier = {
						modifier = suppressed_neutralism
						days = -1
					}
				}
			}
			ai_weight = {
				base = 0
			}
		}
		cancel_suppress_faction = {
			title = "CANCEL_SUPPRESS_FACTION"
			description = "CANCEL_SUPPRESS_FACTION_DESC"
			potential = {
				exists = owner
				parameter:empire = {
					has_modifier = suppressed_neutralism
				}
			}
			effect = {
				remove_modifier = suppressed_faction
				parameter:empire = {
					remove_modifier = suppressed_neutralism
				}
			}
			ai_weight = {
				base = 0
			}
		}
	}
}
