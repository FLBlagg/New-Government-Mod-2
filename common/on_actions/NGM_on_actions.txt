# Triggers when the game starts
on_game_start_country = {
	events = {
		# ポリシーの設定
		ngm_policy.1
		# 青年政治
		ngm_civic.1
		ngm_civic.2
		# 老年政治
		ngm_civic.4
		ngm_civic.5
		# 高度な栄典制度
		ngm_civic.7
		# 繁栄主義者
		ngm_civic.8
		# ゲーム開始後の追加削除不可の国是
		ngm_civic.11
		# 刑罰の金銭代替
		ngm_civic.14
		# 実験的物質の発見
		ngm_civic.16
		# 戦争経済
		ngm_civic.17
		# 異種族解剖
		ngm_civic.20
		# Origin
		ngm_origin.1 # プレーヤを動かすメインイベント # 律儀な古代の緩衝国家
		ngm_origin.3 # プレーヤを動かすメインイベント # 過去への逃走
	}
}

# 帝国のために新しいリーダーを生成し、募集のために利用できるようになります。
# scope: country, from: leader
on_leader_spawned = {
	events = {
		# 青年政治
		ngm_civic.1
		ngm_civic.2
		# 老年政治
		ngm_civic.4
		ngm_civic.5
		# 高度な栄典制度
		ngm_civic.7
		# 特権階級のプライド
		ngm_federation.1
	}
}

# this = country
on_yearly_pulse_country = {
	events = {
		# ゲーム開始後の追加削除不可の国是
		ngm_civic.12
	}
}

# 戦争が始まり、戦争に参加しているすべての国で実行される。
# Root = Country
# From = War
on_war_beginning = {
	events = {
		# 繁栄主義者
		ngm_civic.10
		# 戦争経済
		ngm_civic.18
		# 戦時中イベント
		ngm_government.1
	}
}

# 戦争の終結
# Root = Loser
# From = Main Winner
on_war_ended = {
	events = {
		# 繁栄主義者
		ngm_civic.8
		# 戦争経済
		ngm_civic.19
		# 戦時中イベント
		ngm_government.4
	}
}
