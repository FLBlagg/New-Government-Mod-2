namespace = ngm_federation
country_event = {
	id = ngm_federation.1
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		exists = owner.federation
		owner.federation = {
			has_federation_perk = NGM_PU_TPP
		}
	}
	immediate = {
		every_owned_leader = {
			limit = {
				NOT = {
					has_leader_flag = flag_NGM_PU_TPP_bonus
				}
			}
			set_leader_flag = flag_NGM_PU_TPP_bonus
			set_skill = 2
		}
	}
}
