	id = ngm_origin.3
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		has_origin = origin_ngm_escape_to_the_past
	}
	immediate = {
		if = {
			limit = {
				owner = {
					is_ai = yes
				}
			}
			owner = {
				add_resource = {
					alloys = 2500
				}
			}
		}
		every_owned_planet = {
			add_modifier = {
				modifier = ngm_space_time_continuum
			}
		}
		country_event = {
			id = ngm_origin.4
			days = 3600
		}
	}
}

country_event = {
	id = ngm_origin.4
	hide_window = yes
	location = root
	is_triggered_only = yes
	trigger = {
		has_origin = origin_ngm_escape_to_the_past
	}
	## イベントの効果
	immediate = {
		if = {
			limit = {
				owner = {
					NOR = {
						has_technology = tech_fusion_power
						has_technology = tech_lasers_2
						has_technology = tech_ship_armor_2
						has_technology = tech_thrusters_2
					}
				}
			}
			set_country_flag = ngm_pursuer_1_flag
		}
		else_if = {
			limit = {
				owner = {
					NOR = {
						has_technology = tech_cold_fusion_power
						has_technology = tech_lasers_3
						has_technology = tech_ship_armor_3
						has_technology = tech_thrusters_3
					}
				}
			}
			remove_country_flag = ngm_pursuer_1_flag
			set_country_flag = ngm_pursuer_2_flag
		}
		else_if = {
			limit = {
				owner = {
					NOR = {
						has_technology = tech_antimatter_power
						has_technology = tech_lasers_3
						has_technology = tech_ship_armor_3
						has_technology = tech_thrusters_3
					}
				}
			}
			remove_country_flag = ngm_pursuer_2_flag
			set_country_flag = ngm_pursuer_3_flag
		}
		else_if = {
			limit = {
				owner = {
					NOR = {
						has_technology = tech_zero_point_power
						has_technology = tech_lasers_4
						has_technology = tech_ship_armor_4
						has_technology = tech_thrusters_4
					}
				}
			}
			remove_country_flag = ngm_pursuer_3_flag
			set_country_flag = ngm_pursuer_4_flag
		}
		else_if = {
			limit = {
				owner = {
					NOR = {
						has_technology = tech_dark_matter_power_core
						has_technology = tech_lasers_5
						has_technology = tech_ship_armor_5
						has_technology = tech_dark_matter_propulsion
					}
				}
			}
			remove_country_flag = ngm_pursuer_4_flag
			set_country_flag = ngm_pursuer_5_flag
		}
		else = {
			remove_country_flag = ngm_pursuer_5_flag
			set_country_flag = ngm_pursuer_6_flag
		}
		if = {
			limit = {
				has_country_flag = ngm_pursuer_1_flag
			}
			country_event = {
				id = ngm_origin.5
			}
		}
		else_if = {
			limit = {
				has_country_flag = ngm_pursuer_2_flag
			}
			country_event = {
				id = ngm_origin.6
			}
		}
		else_if = {
			limit = {
				has_country_flag = ngm_pursuer_3_flag
			}
			country_event = {
				id = ngm_origin.7
			}
		}
		else_if = {
			limit = {
				has_country_flag = ngm_pursuer_4_flag
			}
			country_event = {
				id = ngm_origin.8
			}

				id = ngm_origin.9
